returns void
language plpgsql
security definer

function fmtTimeHHMM(t) {
  if (!t) return "09:00";
  const s = String(t);
  return s.length >= 5 ? s.slice(0, 5) : s;
}

function makeTodaySlotIso(timeStr) {
  const [hh, mm] = (timeStr || "09:00").split(":").map(x => parseInt(x, 10));
  const d = new Date();
  d.setHours(Number.isFinite(hh) ? hh : 9, Number.isFinite(mm) ? mm : 0, 0, 0);
  return d.toISOString();
}

create or replace function log_med_intake(
  _patient_id uuid,
  _medication_id uuid,
  _scheduled_at timestamptz,
  _status text
)
returns void
language plpgsql
security definer
set search_path = public
as $$
declare
  my_uid uuid;
  my_email text;
  my_role text;
  effective_role text;
begin
  my_uid := auth.uid();

  if my_uid is null then
    raise exception 'Not authenticated';
  end if;

  select email, role
    into my_email, my_role
  from profiles
  where id = my_uid;

  if my_role = 'admin' then
    effective_role := 'admin';
  elsif my_role = 'caregiver' then
    effective_role := 'caregiver';
  elsif exists (
    select 1
    from patient_partners pp
    where pp.patient_id = _patient_id
      and lower(pp.partner_email) = lower(my_email)
  ) then
    effective_role := 'team_member';
  else
    effective_role := coalesce(my_role, 'caregiver');
  end if;

  insert into medication_logs (
    patient_id,
    medication_id,
    scheduled_at,
    status,
    recorded_by,
    recorded_role,
    created_at,
    updated_at
  )
  values (
    _patient_id,
    _medication_id,
    _scheduled_at,
    _status,
    my_uid,
    effective_role,
    now(),
    now()
  )
  on conflict (patient_id, medication_id, scheduled_at)
  do update set
    status = excluded.status,
    recorded_by = excluded.recorded_by,
    recorded_role = excluded.recorded_role,
    updated_at = now();
end;
$$;

Ctrl+V работает только если курсор стоит в поле ввода.

Например:

    в SQL Editor нужно сначала кликнуть мышкой в окно SQL

    потом нажать Ctrl + V.

2️⃣ Копирование не произошло

Иногда текст не был выделен.

Правильно так:

    Выдели текст мышкой

    Нажми Ctrl + C

    Вставь Ctrl + V

3️⃣ Раскладка клавиатуры

Иногда Windows воспринимает комбинацию неправильно.

Попробуй:

Ctrl + Insert  → копировать
Shift + Insert → вставить

Это работает даже когда Ctrl+C сломан.
4️⃣ Браузер блокирует вставку

В некоторых браузерах (особенно Edge) SQL Editor иногда блокирует вставку.

Попробуй:

Правый клик → Paste
5️⃣ Залипшая клавиша Ctrl

Проверь:

Нажми просто букву C.
Если получается Ctrl+C-поведение, значит Ctrl залипла.

✅ Самый быстрый способ для SQL Editor:

    выдели текст

    правый клик → Copy

    в SQL Editor правый клик → Paste
